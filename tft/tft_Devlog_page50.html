14:28:22MISCFor those interested, below is the script I used to generate the weapons, and I use a CLI tool for Lua to get the results ;)
WeaponsFor = 
      {Jake = { EQP_STAT_Power={2}, ITM_ShopPrice = {250,"Dub"} },
       Dandor = { EQP_STAT_Power={4}, ITM_ShopPrice = {2000,"Dub"} },
       HandoStillor = { EQP_STAT_Power={1}, EQP_STAT_Intelligence={2}, ITM_ShopPrice = {300,"Dub"} }
       }
ArmorFor = {Jake = {EQP_STAT_Endurance={2}, ITM_ShopPrice = {100,"Dub"} },
            Marrilona = {EQP_STAT_Intelligence={1},EQP_STAT_Endurance={1}, EQP_STAT_Resistance={3}, ITM_ShopPrice={125,'Dub'} },
            HandoStillor = {EQP_STAT_Endurance={1},EQP_STAT_Resistance={1}, ITM_ShopPrice = {112,'Dub'} }
           }
            
Workout = {WP = WeaponsFor,AR = ArmorFor}

upper = string.upper

level=6

for lv=1,level do
    for WOK,WOA in pairs(Workout) do
        for EQK,EQV in pairs(WOA) do
            print("Rec: EQP_"..EQK.."_"..WOK..lv)
            print("	ITM_Type = "..({WP='Weapon',AR='Armor'})[WOK])
            print("	ITM_EQP_For = "..EQK)
            for STK,STV in pairs(EQV) do
                id = WOK..EQK..STK
                value = value or {}
                value[id] = value[id] or 0;
                ({ Plus = function() value[id] = value[id] + STV[1] end,
                   Dub  = function() if value[id]==0 then value[id]=STV[1] else value[id] = value[id] + value[id] end end})[STV[2] or 'Plus']()
                print("	"..STK.." = "..value[id])   
            end
        end    
    end
end
