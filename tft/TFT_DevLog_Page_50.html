13:17:55TECHNO
Now some people claim well coded code contains NO "if"-statements at all. Why people claim something that stupid I don't know but at least they can be happy about the way the Grombo town has been scripted
function NPC_INNKEEPER()
  Inn()
end

function NPC_Sandra()
  MapText("SANDRA_" ..
            ({
               [false] = "A",
               [true] = "B"
            })[Done('&DONE.GROMBO.SPOKEN.SANDRA')]
         )
end

function QuendorGive()
   Var.D("$WMCHAT","QUENDOR")
   ItemGive("ZZKEY_GAIA")
end   

function NPC_Quendor()
       local ldone = Done("DONE.GROMBO.SPOKEN.QUENDOR")
       MapText("QUENDOR_"..upper(sval(ldone)));
       ({ [true] = Nothing, [false]=QuendorGive })[ldone]()       
end

function GALE_OnLoad()
    ZA_Enter('ToQuendor',function() GoToLayer('quendor',"Start") end)
    ZA_Enter("ToTown",function() GoToLayer('town',"Einde") end)
end
